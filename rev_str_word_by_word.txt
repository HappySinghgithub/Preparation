                6. Reverse string word by word
#include<stdio.h>
#include<conio.h>
#include<string.h>
void main()
{
    char a[50],b[50];
    int i=0,j=0,p;
    printf("Enter the string: ");
    gets(a);
    for(i=strlen(a)-1;i>=0;i--)
    {
        if(a[i]!=' ')
        {
            b[j]=a[i];
            j++;
        }
        else
        {
           b[j]='\0';
            for(p=strlen(b)-1;p>=0;p--)
            {
                printf("%c",b[p]);
            }
            j=0;
            printf(" ");
        }
    }
    b[j]='\0';
    for(p=strlen(b)-1;p>=0;p--)
    {
        printf("%c",b[p]);
    }
}
